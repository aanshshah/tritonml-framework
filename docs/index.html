Quick Start

Get up and running in minutes

1

Install TritonML

pip install tritonml
2

Deploy Your Model

from tritonml import TritonModel

# Load any HuggingFace model
model = TritonModel.from_huggingface(
    "bert-base-uncased",
    task="text-classification"
)

# Optimize and deploy
model.quantize()  # 4x compression
client = model.deploy()
3

Make Predictions

# Single prediction
result = model.predict("This movie is amazing!")

# Batch predictions
texts = ["Great product!", "Terrible service"]
results = model.predict(texts)

# Get probabilities
probs = model.predict_proba(texts)

More Examples

from tritonml.tasks import EmotionClassifier

# Specialized emotion detection model
model = EmotionClassifier.from_pretrained()
model.deploy()

# Analyze emotions
texts = [
    "I'm so angry about this!",
    "This is the best day ever!",
    "Things will get better soon.",
    "I feel so sad and lonely."
]

emotions = model.predict(texts)
for text, emotion in zip(texts, emotions):
    print(f"'{text}' → {emotion}")
# Deploy a model from CLI
tritonml deploy bert-base-uncased --task text-classification

# Make predictions
tritonml predict my-model "Hello world!" --server localhost:8000

# List deployed models
tritonml model list --server localhost:8000

# Benchmark performance
tritonml benchmark my-model --batch-sizes 1,8,16,32
from tritonml.deploy.docker import create_deployment_package

# Generate complete Docker deployment
create_deployment_package(
    model_name="emotion-classifier",
    output_path="./deploy"
)

# This creates:
# - Dockerfile
# - docker-compose.yml
# - client_example.py
# - README.md
# Benchmark your model
results = model.benchmark(
    test_inputs=["sample"] * 100,
    batch_sizes=[1, 8, 16, 32]
)

# Results:
# batch_1: 12.5ms latency, 80 samples/sec
# batch_8: 15.2ms latency, 526 samples/sec  
# batch_16: 18.7ms latency, 855 samples/sec
# batch_32: 25.3ms latency, 1264 samples/sec
